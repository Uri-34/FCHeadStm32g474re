88enum Commands
89{
90// Установки
91 NU = 0x00, // ничего
92 GAMSET = 0x26, // выбор таблицы гаммы (Gamma Set).
93 INVOFF = 0x20, // отключить инверсию цвета.
94 INVON = 0x21, // включить инверсию цвета.
95 RAMCTRL = 0xB0, // управление RAM (опционально, зависит от версии ST7789).
96 RGBCTRL = 0xB1, // управление форматом RGB (часто не используется).
97 PORCTRL = 0xB3, // управление частотой кадров (Partial Mode On + Frame Rate).
98 IDMOFF = 0xB6, // выключить idle mode.
99 IDMON = 0xB7, // включить idle mode.
100 PWCTRL1 = 0xC0, // управление питанием (Power Control 1).
101 PWCTRL2 = 0xC1, // power Control 2.
102 PWCTRL3 = 0xC2, // power Control 3.
103 PWCTRL4 = 0xC3, // power Control 4.
104 PWCTRL5 = 0xC4, // power Control 5.
105 VMCTRL1 = 0xC5, // управление напряжением (VCOM Control 1).
106 VCOMS = 0xC6, // установка напряжения VCOM (VCOM Setting).
107 VCOMVR = 0xC7, // диапазон VCOM (VCOM Voltage Range).
108 NVMPKEY = 0xD0, // ключ доступа к NVM (если есть).
109 PGAMCTRL = 0xE0, // настройка положительной гаммы (Positive Gamma Control). Принимает 14-16 параметров.
110 NGAMCTRL = 0xE1, // настройка отрицательной гаммы (Negative Gamma Control).
111
112// Комманды
113 SWRESET = 0x01, // программный сброс (Software Reset). Обязательно после питания.
114 SLPIN = 0x10, // вход в спящий режим (Sleep In).
115 SLPOUT = 0x11, // выход из спящего режима (Sleep Out). Требует задержки ~120 мс перед следующей командой.
116 DISPON = 0x29, // включить дисплей (Display On).
117 DISPOFF = 0x28, // выключить дисплей (Display Off).
118 CASET = 0x2A, // установка горизонтального окна (Column Address Set). Принимает Xstart, Xend (по 2 байта каждый).
119 RASET = 0x2B, // установка вертикального окна (Row Address Set). Принимает Ystart, Yend.
120 RAMWR = 0x2C, // запись в память дисплея (Memory Write). После этой команды идут пиксельные данные RGB565.
121 RAMRD = 0x2E, // чтение из памяти дисплея (Memory Read). Редко используется в SPI-режиме.
122 MADCTL = 0x36, // Управление ориентацией и порядком пикселей (Memory Access Control).
123 // Определяет поворот экрана, RGB/BGR, вертикаль/горизонталь.
124 COLMOD = 0x3A, // Установка цветового формата (Interface Pixel Format).
125
126 RGB565 = 0x55, // Обычно 0x55 — 16 бит на пиксель (RGB565)
127
128// Частичное обновление
129 VSCSAD = 0x37, // вертикальное смещение начала отображения (Vertical Scroll Start Address). Для аппаратного скролла.
130 TCRTL = 0x38, // управление Tearing Effect (для синхронизации с кадром).
131 TEON = 0x39, // включить Tearing Effect.
132 TCTRL = 0x35, // управление режимом Tearing Effect.
133 VSCRDEF = 0x33, // определение области вертикального скролла.
134
135// Режимы
136 RDDID = 0x04, // чтение ID дисплея (Read Display ID).
137 RDDPM = 0x09, // чение статуса питания (Read Display Power Mode).
138 RDDMADCTL = 0x0A, // чтение MADCTL.
139 RDDCOLMOD = 0x0B, // чтение COLMOD.
140 RDDIM = 0x0C, // чтение Image Mode.
141 RDDSM = 0x0D, // чтение Signal Mode.
142 RDDSDR = 0x0F, // чтение Self-Diagnostic Result.
143 RDID1 = 0xDA, // чтение ID1 (производитель).
144 RDID2 = 0xDB, // чтение ID2.
145 RDID3 = 0xDC, // чтение ID3.
146 RAMCTRL = 0xDE, // чтение RAM control (если поддерживается).
147 RGBCTRL = 0xDF // чтение RGB control.
148};
