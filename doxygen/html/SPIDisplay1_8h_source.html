90 static void flush(lv_disp_drv_t* disp, const lv_area_t* area, lv_color_t* color)
91 {
92 SPIDisplayDriver* driver = static_cast<SPIDisplayDriver*>(disp->user_data);
93 driver->flush_impl(area, color);
94 }
95
96 void flush_impl(const lv_area_t* area, lv_color_t* color)
97 {
98 // Твой код отрисовки
99 _dc = 1;
100 _cs = 0;
101
102 uint32_t w = (area->x2 - area->x1 + 1);
103 uint32_t h = (area->y2 - area->y1 + 1);
104
105 // Установка окна
106 setAddrWindow(area->x1, area->y1, w, h);
107
108 // Отправка пикселей
109 for (uint32_t i = 0; i < w * h; i++) {
110 uint16_t px = color[i].full;
111 _spi->write(px >> 8);
112 _spi->write(px);
113 }
114
115 _cs = 1;
116
117 // ВАЖНО: сигнализируем LVGL, что готовы
118 lv_disp_flush_ready(disp);
119 }
120
121 bool init();
122
123public:
124 SPIDisplayDriver(SPIBus* spi, Size2D &size, PinName cs, PinName dc, PinName rst, Object *parent = nullptr)
125 : Driver("", 0, parent),
126 _spi{spi},
127 _cs{cs},
128 _dc{dc},
129 _rst{rst},
130 _size{size}
131 {
132 if(init() && (_drawBuffer = new lv_color_t[size._w * size._h]))
133 {
134 set(Ready);
135 }
136 }
137
138 ~SPIDisplayDriver()
139 {
140 if(_drawBuffer)
141 {
142 delete _drawBuffer;
143 }
144 }
145
146 bool init()
147 {
148 _spi->format(8, 0);
149 _spi->frequency(40000000); // 40 МГц
150
151 // Инициализация дисплея
152 _rst = 0;
153 ThisThread::sleep_for(100ms);
154 _rst = 1;
155 ThisThread::sleep_for(120ms);
156
157 auto writeCmd = [&](uint8_t cmd)
158 {
159 _dc = 0;
160 _cs = 0;
161 _spi->write(cmd);
162 _cs = 1;
163 };
164
165 auto writeData = [&](const uint8_t* data, size_t len)
166 {
167 _dc = 1;
168 _cs = 0;
169 for(size_t count = 0; count < len; count++)
170 {
171 _spi->write(data[count]);
172 }
173 _cs = 1;
174 };
175
176 // Пример инициализации ST7789
177 writeCmd(0x36); // Memory Access Control
178 uint8_t val = 0x60;
179 writeData(&val, 1);
180
181 writeCmd(0x3A); // Pixel Format
182 val = 0x05; // 16 бит
183 writeData(&val, 1);
184
185 writeCmd(0x21); // Inversion ON
186 writeCmd(0x11); // Sleep Out
187 ThisThread::sleep_for(120ms);
188 writeCmd(0x29); // Display ON
189
190 // Инициализация LVGL v8
191 lv_init();
192
193 // Настройка буфера
194 static lv_disp_draw_buf_t draw_buf;
195 lv_disp_draw_buf_init(&draw_buf, _drawBuffer, nullptr, _size._w * _size._h);
196
197 // Настройка драйвера

199 lv_disp_drv_init(&disp_drv);
200 disp_drv.draw_buf = &draw_buf;
201 disp_drv.flush_cb = flush;
202 disp_drv.hor_res = _size._w;
203 disp_drv.ver_res = _size._h;
204 disp_drv.sw_rotate = 1;
205 disp_drv.rotated = LV_DISP_ROT_90;
206 disp_drv.user_data = this;
207 lv_disp_drv_register(&disp_drv);
